pipeline {
        agent {
            label {
            label 'built-in'
            customWorkspace "/mnt/project"
            }
        }
        
		stages {
		       stage ("install httpd") {
			          steps {
					      sh "yum install httpd -y"
						  sh "systemctl start httpd"
					  }
			   }
			   
			   stage ("create index file") {
			          steps {
			               sh 'echo "<h1> welcome to my index page </h1>" > index.html'
					  }
			   }
			   
			   stage ("deploy") {
					 steps {
					       sh "cp index.html /var/www/html/"
					 }
			   }
			   
			   stage ("permission") {
			         steps {
					       sh "chmod 777 /var/www/html/index.html"
						   sh "systemctl restart httpd"
					 }
			   }
			   
		
		}
}
